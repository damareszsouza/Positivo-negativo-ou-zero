Exercício : Positivo, Negativo ou Zero
Gabarito do Exercício 
programa

{

   funcao inicio(){

      inteiro valor

      escreva ("Informe o valor: ")

      leia(valor)

      se(valor == 0) {

         escreva ("O número é zero")

      }

      senao{

         se(valor > 0) {

            escreva ("O número é positivo")

         }

         senao{

            escreva ("O número é negativo")

         }

      }

   }

}
